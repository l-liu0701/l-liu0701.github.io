# 自动生成测试数据

本实验使用的RDBMS为MySQL，OS为macOS。

数据量 数据生成方式
小 通过INSERT向表中插入数据
大 在内存表中创建数据，然后一次性插入测试表中^1

# 1. 内存表

# 1.1 内存表特征

MySQL的内存表的表结构创建在磁盘上，数据存放在内存中。

内存表结构/内存表数据 存储位置 扩展名
表结构 磁盘 .frm
数据 内存 /
  • MySQL启动时：表结构存在，表数据存在
  • MySQL重启后：表结构存在，表数据消失^2

# 1.2 内存表优点

内存表使用哈希散列索引将数据保存在内存中，因此具有极快的速度。

# 1.3 内存表的使用

heap[^4]对所有用户的连接是可见的，这使得它非常适合做缓存。

# 1.4 创建内存表

create table mem_table(
    id int(10),
	  num int(10)
) engine=memory;		-- 创建内存表时需要选择内存模式memory

# 2. 自定义函数

自定义函数的两个必要条件：

  • 参数
  • 返回值

# 2.1 创建自定义函数

create function function_name
returns
{string|integer|real|decimal}
routine_body

# 2.1.1 函数体

  • 函数体由合法的SQL语句构成
    • 函数体可以是简单的SELECT或INSERT语句
  • 函数体如果为复合结构则使用BEGIN...END语句
    • 复合结构可以包含声明、循环、控制结构

delimiter //
create function rand_string(n int)
returns varchar(255)
begin
    declare chars_str varchar(100) default 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
    declare res_str varchar(255) default '';
    declare i int default 0;
    while i < n do
        set res_str = concat(res_str, substring(chars_str , FLOOR(1 + RAND()*62 ),1)); 
        set i = i + 1;
    end while;
    return res_str;
end //
delimiter ;

# 3. 存储过程[^3]

# 3.1 优点

  • 存储过程可封装，并隐藏复杂的商业逻辑
  • 存储过程可以回传值，并可以接受参数
  • 存储过程无法使用SELECT指令来运行，因为它是子程序，与查看表，数据表或用户定义函数不同
  • 存储过程可以用在数据检验，强制实行商业逻辑等

# 3.2 缺点

  • 由于支持的编程语言不同，存储过程往往定制化于特定的数据库上
  • 存储过程的性能调校于撰写受限于各种数据库系统

# 3.3 创建存储过程

CREATE
    [DEFINER = { user | CURRENT_USER }]
　PROCEDURE sp_name ([proc_parameter[,...]])
    [characteristic ...] routine_body
 
proc_parameter:
    [ IN | OUT | INOUT ] param_name type
 
characteristic:
    COMMENT 'string'
  | LANGUAGE SQL
  | [NOT] DETERMINISTIC
  | { CONTAINS SQL | NO SQL | READS SQL DATA | MODIFIES SQL DATA }
  | SQL SECURITY { DEFINER | INVOKER }
 
routine_body:
　　Valid SQL routine statement
 
[begin_label:] BEGIN
　　[statement_list]
　　　　……
END [end_label]

# 3.4 例子

# 3.4.1 创建存储过程

-- 次存储过程调用了<2.4.2.a>中的rand_string()自定义函数
-- cd ~/tmp
-- touch test1.sql
-- code test1.sql
delimiter //
create procedure add_user_memory(n int, len int)
begin
    declare i int default 1;
    while (i <= n) do
        insert into user_mem(username, password) values(rand_string(len), rand_string(len));
        set i = i + 1;
    end while;
end //
delimiter ;

# 3.4.2 执行存储过程的sql语句

source ~/tmp/test1.sql

# 3.4.3 调用存储过程

call add_user_memory(200, 20);
